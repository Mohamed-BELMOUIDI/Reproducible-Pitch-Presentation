# Example of how simple the analysis can be
sample_data <- c(1, 2, 3, 4, 5)
cat("Basic Statistics:\n")

## Basic Statistics:

cat("Mean:", mean(sample_data), "\n")

## Mean: 3

cat("Median:", median(sample_data), "\n")

## Median: 3

cat("SD:", sd(sample_data), "\n")

## SD: 1.581139
